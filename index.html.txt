    function initImageUpload() {
        multiImagesInput.addEventListener('change', handleImageSelection);

    function handleImageSelection(e) {
        const files = Array.from(e.target.files);
        if (files.length === 0) return;
        if (currentImageFiles.length + files.length > MAX_IMAGES) {
            alert(`لا يمكنك إضافة أكثر من ${MAX_IMAGES} صور`);
            multiImagesInput.value = '';
            return;
        }
        for (let file of files) {
            const exists = currentImageFiles.some(f => f.name === file.name && f.size === file.size);
            if (!exists) currentImageFiles.push(file);
        }
        updateImagePreview();
        multiImagesInput.value = '';
    }
    function updateImagePreview() {
        imagesPreviewContainer.innerHTML = '';
        currentImageFiles.forEach((file, index) => {
            const reader = new FileReader();
            reader.onload = function(e) {
                const div = document.createElement('div');
                div.className = 'preview-item';
                div.innerHTML = `<img class="preview-img" src="${e.target.result}"><button class="remove-img-btn" onclick="window.removeImage(${index})">×</button>`;
                imagesPreviewContainer.appendChild(div);
            };
            reader.readAsDataURL(file);
        });
        imageCounterSpan.textContent = `${currentImageFiles.length} / ${MAX_IMAGES} صورة`;
    }
    window.removeImage = function(index) {
        currentImageFiles.splice(index, 1);
        updateImagePreview();
    };
    function resetImages() {
        currentImageFiles = [];
        updateImagePreview();
        multiImagesInput.value = '';

    // ------------------- رفع الصور إلى Cloudinary -------------------
    async function uploadImageToCloudinary(file) {
        const formData = new FormData();
        formData.append('file', file);
        formData.append('upload_preset', UPLOAD_PRESET);
        formData.append('folder', 'marketplace');
        const response = await fetch(`https://api.cloudinary.com/v1_1/${CLOUD_NAME}/image/upload`, { method: 'POST', body: formData });
        const data = await response.json();
        if (!response.ok) throw new Error(data.error?.message);
        return data.secure_url;
    }

        const loadingMsg = showLoadingMessage('جاري رفع الصور...');
        try {
            let uploadedUrls = [];
            for (let file of files) { const url = await uploadImageToCloudinary(file); uploadedUrls.push(url); }
            await db.collection('marketplace_items').add({
                title, images: uploadedUrls, fullDescription: fullDesc, condition, quantity, requestType, price, location,
                sellerName: fullName, sellerPhone: phone, status: 'pending', createdAt: firebase.firestore.FieldValue.serverTimestamp(),
                reservedBy: null, sellerId: currentUser.nationalId
            });
            hideLoadingMessage(loadingMsg);
            showMessage(msgDiv, 'تم إرسال الإعلان للمراجعة', 'success');
            document.getElementById('itemTitle').value = ''; document.getElementById('fullDescription').value = '';
            document.getElementById('location').value = ''; resetImages(); document.getElementById('price').value = '0';
            await loadListings(); renderUserListings();
        } catch (err) { hideLoadingMessage(loadingMsg); showMessage(msgDiv, 'فشل: ' + err.message, 'error'); }
    }

    // ------------------- حجز السلعة المجانية -------------------
    async function reserveItem(itemId) {
        const item = allListings.find(i => i.id === itemId);
        if (!item || item.price !== 0 || item.status !== 'approved') { alert('غير متاح'); return; }
        const name = prompt('أدخل اسمك الكامل:'); if (!name) return;
        const phone = prompt('أدخل رقم هاتفك:'); if (!phone) return;
        await db.collection('marketplace_items').doc(itemId).update({ status: 'reserved', reservedBy: { name, phone, reservedAt: new Date().toISOString(), reservedByUserId: currentUser.nationalId } });
        alert('تم الحجز');
        await loadListings(); renderUserListings();
        if (currentUser.role === 'admin' || currentUser.role === 'supervisor') loadAdminListings('reserved');
    }

    window.updateItemStatus = async function(itemId, newStatus) {
        await db.collection('marketplace_items').doc(itemId).update({ status: newStatus });
        alert(`تم تغيير الحالة إلى ${newStatus}`);
        const activeTab = document.querySelector('.admin-tab.active');
        const filter = activeTab ? activeTab.getAttribute('data-status') : 'pending';
        await loadAdminListings(filter);
        await loadListings();
        renderUserListings();
    };
    window.deleteItem = async function(itemId) {
        if (!confirm('حذف نهائي؟')) return;
        await db.collection('marketplace_items').doc(itemId).delete();
        alert('تم الحذف');
        const activeTab = document.querySelector('.admin-tab.active');
        const filter = activeTab ? activeTab.getAttribute('data-status') : 'pending';
        await loadAdminListings(filter);
        await loadListings();
        renderUserListings();
    };

    function escapeHtml(str) { if (!str) return ''; return str.replace(/[&<>]/g, m => ({ '&': '&amp;', '<': '&lt;', '>': '&gt;' }[m] || m)); }
    function showMessage(el, text, type) { el.textContent = text; el.className = `message ${type}`; el.classList.remove('hidden'); setTimeout(() => el.classList.add('hidden'), 5000); }
    function showLoadingMessage(msg) { const div = document.createElement('div'); div.id = 'loadingOverlay'; div.style.cssText = 'position:fixed; top:0; left:0; width:100%; height:100%; background:rgba(0,0,0,0.7); display:flex; justify-content:center; align-items:center; z-index:10000;'; div.innerHTML = `<div style="background:white; padding:20px; border-radius:20px;"><i class="fas fa-spinner fa-spin"></i> ${msg}</div>`; document.body.appendChild(div); return div; }
    function hideLoadingMessage(el) { if (el) el.remove(); }
